create table asociacion(
    codigo char(4) primary key, 
    nombre varchar(40),
    destacados number(2)
)

create table entrenador(
    nif char(9) primary key,
    nombre varchar(40),
    nivel number(2) not null,
    asociacion char(4) not null references asociacion
)

create table equipo(
    codigo char(4) primary key,
    nombre varchar(40),
    nivel number(2) not null,
    asociacion char(4) not null references asociacion
)

CREATE TABLE libre(
  entrenador char(9) references entrenador,
  nombre varchar(40),
  total number(3),
  primary key(entrenador)
);

CREATE TABLE responsable(
  equipo char(4) references equipo,
  entrenador char(9) references entrenador,
  primary key(equipo,entrenador)
);

create table jugar(
    equipo1 char(4) references equipo,
    equipo2 char(4) references equipo,
    goles1 number(2),
    goles2 number(2),
    primary key(equipo1, equipo2)
);
