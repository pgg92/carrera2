create table deportista(
    nif char(4) primary key, 
    nombre varchar(40),
    categoria char(40)
);

create table prueba(
    codigo char(4) primary key,
    campeonato char(4) references campeonato,
    fecha date,
    lugar varchar(40),
    localidad char(40),
    provincia char(40)
);

create table campeonato ( 
  codigo char(4) primary key,
  nombre char(30)
  );

create table participar (
  nif char(4) references deportista,
  codigo char(4)references campeonato,
  puntos number,
  primary key(nif,codigo)
);
